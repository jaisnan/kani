4 // This test checks that kani injects a reachability check for
5 // remainder-by-zero checks and that it reports ones that are unreachable.
6 // The check in this test is reachable, so should be reported as SUCCESS
7
8 fn rem(x: u16, y: u16) -> u16 {
9 if y != 0 { x % y } else { 0 }
10 }
11
12 #[kani::proof]
13 fn main() {
14 rem(11, 3);
15 }
