4 //! Test that the abort() function is respected and nothing beyond it will execute.
5
6 use std::process;
7
8 #[kani::proof]
9 fn main() {
10 for i in 0..4 {
11 if i == 1 {
12 // This comes first and it should be reachable.
13 process::abort();
14 }
15 if i == 2 {
16 // This should never happen.
17 process::exit(1);
18 }
19 }
20 assert!(false, "This is unreachable");
21 }
