4 // Check the location of the assert statement when using debug_assert!
5 // This currently requires the old format since the new format doesn't include
6 // line numbers https://github.com/model-checking/kani/issues/918
7 // Also disable reachability checks because they add annotations to each
8 // assert's description which would be visible with the old output format
9 // kani-flags: --output-format old --no-assertion-reach-checks
10 #[kani::proof]
11 fn main() {
12 for i in 0..4 {
13 debug_assert!(i > 0, "This should fail and stop the execution");
14 assert!(i == 0, "This should be unreachable");
15 }
16 }
